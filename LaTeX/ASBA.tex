	\newcommand*{\idxdictionary}[2]{%   nur für Indices
		\extendtheindex{}{%             am Anfang von '\printindex'
			\likesection{#2}
			Index und Symbolverzeichnis enthalten keine gemeinsamen Einträge.

			\idxpreambel
		}{}{}

		\Chead{#2}%                     Kopfzeile Mitte = <Verzeichnisname>
