\newcommand*{\Datenbank}[5]{% 2(#1) x 2(#2+#3) = (-#1,-#2-#3),(+#1,+#2+#3)
	% Halbmesser x, Halbmesser y, halbe Höhe, Name1, Name2 - Ursprung in der Mitte
	\put(0.0,-#3){
		\qbezier(-#1,0.0)(-#1,-#2)(0.0,-#2)
		\qbezier(+#1,0.0)(+#1,-#2)(0.0,-#2)
	}
	\put(0,0){\Line(-#1,-#3)(-#1, #3)}
	\put(0,0){\Line( #1,-#3)( #1, #3)}
	\put(0.0,#3){
		\qbezier(-#1,0.0)(-#1,-#2)(0.0,-#2)
		\qbezier( #1,0.0)( #1,-#2)(0.0,-#2)
		\qbezier(-#1,0.0)(-#1, #2)(0.0, #2)
		\qbezier( #1,0.0)( #1, #2)(0.0, #2)
	}
	\makebox(0,  0){\ImageFt{#4}}
	\makebox(0,-#3){\ImageFt{#5}}
}
